    # ajax

    # 介绍

    ajax:异步程序

    # 请求状态码:

    • ​ 0: 请求未初始化
    • ​ 1: 服务器连接已建立
    • ​ 2: 请求已接收
    • ​ 3: 请求处理中
    • ​ 4: 请求已完成，且响应已就绪

    # 协议状态码:

    • ​ 1XX(临时相应）
    • ​ 2XX(成功)
    • ​ 200(成功）请求成功
    • ​ 3xx（重定向）
    • ​ 4xx（请求错误）
    • ​ 400（错误请求）： 服务器不理解请求的语法；
    • ​ 404（未找到指定文件） 路径问题
    • ​ 5XX（服务器错误）
    • ​ 500（服务器内部错误）： 服务器遇到错误，无法完成请求；

    # ajax post请求

     btn.onclick = function () {
            // 1.创建ajax对象
            var xhr = new XMLHttpRequest();
           
            // 2.发送请求 open(method,url)
            xhr.open("post", "links/2.post.php")
            
            // post 请求  设置请求头
            xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
            // 3.发送数据
            xhr.send("username=ahua&age=20")
          
            // 4.监听响应结果
            xhr.onreadystatechange = function () {
                // 1.请求状态码  2.协议状态码
                if(xhr.readyState == 4 && xhr.status == 200){
                    // responseText response 接口中的数据
                    res.innerHTML = xhr.response
                }else{
                    console.log("做请求失败的操作")
                }
            }


        }

    # 封装ajax

      // ajax 封装
        function ajax(method, url, obj, callback) {
            // username=awei&age=20
            // 处理对象格式数据 变成字符串
            var arr = []
            for (var key in obj) {
                arr.push(key + "=" + obj[key])
            }
            var data = arr.join("&");
            // 1.创建ajax对象
            var xhr = null;
            // 判断高低版本浏览器
            if (window.XMLHttpRequest) {
                xhr = new XMLHttpRequest();//IE7+, Firefox, Chrome, Opera, Safari
            } else {// code for IE6, IE5
                xhr = new ActiveXObject("Microsoft.XMLHTTP");
            }
            // 判断请求方式 是否为get请求
            url = method == "get" ? url + "?" + data : url;

            //2.发送请求 
            xhr.open(method, url)

            // 3.发送数据
            if (method == "get") {
                // 说明是get请求  无需发送数据
                xhr.send(null)
            } else {
                // 设置请求头
                xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
                // post请求 需发送数据
                xhr.send(data)
            }
            // 4.监听响应结果
            xhr.onreadystatechange = function () {
                // 判断 是否请求成功
                if (xhr.readyState == 4 && xhr.status == 200) {
                    callback && callback(xhr.response)
                }
            }


        }
